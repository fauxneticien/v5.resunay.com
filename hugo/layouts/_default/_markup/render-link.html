Only affects Markdown links — shortcodes/raw HTML manage their own. */ -}} {{- $u := urls.Parse .Destination -}} {{- $siteHost := (urls.Parse site.BaseURL).Host -}} {{- $external := and $u.Host (ne $u.Host $siteHost) -}} {{ .Text | safeHTML }} {{- /* No trailing newline: the hook's output is emitted inline, so a trailing newline collapses to a stray space before following punctuation (e.g. a link ending a sentence would render as "text ."). This comment's `{{-` trims the preceding newline; `-}}` trims the file's final newline. */ -}}
